class Solution {
    public String generateTheString(int n) {
        StringBuilder sb = new StringBuilder();
        for (int i = 0; i < n; i++) {
            sb.append("a");
        }
        if (n % 2 == 0) {
            sb.deleteCharAt(0);
            sb.append("b");
        }
        return sb.toString();
    }
}
/*
    解题思路：
        0.单独分为奇数和偶数处理
        1.如果是偶数，那么n可以分为1和n-1；如果是奇数，那么n可以直接是n个相同的字符
        2.注意，如果n为1，那么就单独处理
 */
